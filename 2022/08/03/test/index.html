WGCNA加权基因共表达网络分析(一)

WGCNA

加权基因共表达网络分析 (WGCNA, Weighted correlation network analysis)是用来描述不同样品之间基因关联模式的系统生物学方法，可以用来鉴定高度协同变化的基因集, 也可将模块与样本表型进行关联，鉴定潜在的生物标志物或治疗靶点。官方推荐样本数尽量不小于15个，数据格式统一即可，即FPKM、TPM都可以做WGCNA。

基于R语言的WGCNA分析

1.安装WGCNA及其依赖包

install.packages("WGCNA")
#根据提示安装所需依赖，这里直接安装成功

2.数据处理
基因表达矩阵，行为基因，列为样本

#导入数据
options(stringsAsFactors = FALSE)
femData<-read.csv("Pp_gene_fpkm.csv",row.names = 1)
#简单看下
dim(femData)
names(femData)
##数据过滤，去除低表达或变化不大的基因
#对基因取中值绝对偏差值
exp.mad<-apply(femData,1,mad)
## 筛选中位绝对偏差前75%的基因，至少MAD大于0.01，筛选后会降低运算量，也会失去部分信息，也可不做筛选，使MAD大于0即可
femData.filtered <- femData[which(exp.mad > 
                                max(quantile(exp.mad, probs=seq(0, 1, 0.25))[2],0.01)),]
dim(femData.filtered)
# 转换为样品在行，基因在列的矩阵
datExpr0<-as.data.frame(t(femData.filtered))
#检测缺失值
gsg<-goodSamplesGenes(datExpr0,verbose = 3)
gsg$allOK
#如果gsg$allOk为FALSE，继续进行过滤：
if (!gsg$allOK) {
datExpr0<-datExpr0[gsg$goodSamples,gsg$goodGenes]  
}
## 样本聚类查看是否有离群样品
sampleTree = hclust(dist(datExpr0), method = "average")
#绘制聚类树
plot(sampleTree, main = "Sample clustering to detect outliers", 
     sub="", xlab="",cex = 0.6,cex.lab = 1,
     cex.axis=1,cex.main = 1.2)
#读入表型数据，行为样本，列为表型
traitData<-read.csv("ClinicalTraits.csv",row.names=1)
dim(traitData)
names(traitData)
#对样本重新聚类
sampleTree2<-hclust(dist(datExpr0), method = "average")
#使用热图展示表型
traitColors<-numbers2colors(traitData,signed = FALSE)
#组合
plotDendroAndColors(sampleTree2,traitColors,groupLabels = names(traitData),main="Sample clustering to detect outliers and trait heatmap")
#保存临时数据用于后续分析
save(datExpr0,traitData,file = "WGCNA_01_data.RData")
rm(list=ls())

2.网络构建与模块划分

##2.1软阈值筛选
load(file = "WGCNA_01_data.RData")
#列出环境中的变量
ls()
#选择soft thresholding powers（软阈值），进行网络拓扑分析
powers <- c(c(1:10), seq(from=12,to=20,by=2))
#调用网络拓扑分析函数，计算每个power的值（软阈值，power β）对应的网络拓扑性质
sft<-pickSoftThreshold(datExpr0,powerVector=powers,verbose=5)
sft
#提取作图数据
x1<- sft$fitIndices[,1]
y1<- -sign(sft$fitIndices[,3])*sft$fitIndices[,2]
y2<- sft$fitIndices[,5]
#对拟合结果进行可视化
sizeGrWindow(9,5)
par(mfrow=c(1,2))
cex1=0.9
plot(x1,y1,type="n",
     xlab="Soft Threshold (power)",
     ylab="Scale Free Topology Model Fit,signed R^2",main = paste("Scale independence"))
text(x1, y1,labels=powers,cex=cex1,col="red")
# 筛选标准。R-square=0.85?
abline(h=0.85,col="red")
# Soft threshold与平均连通性
plot(x1, y2,
     xlab="Soft Threshold (power)",ylab="Mean Connectivity", type="n",
     main = paste("Mean connectivity"))
text(x1, y2, labels=powers, cex=cex1, col="red")
power = 14
power
#########经验power (无满足条件的power时选用)# 无向网络在power小于15或有
#网络power小于30内，没有一个power值可以使无标度网络图谱结构R^2达到0.8，
#平均连接度较高如在100以上，可能是由于；部分样品与其他样品差别太大。这可
#能由批次效应、样品异质性或实验条件对表达影响太大等造成。可以通过绘制样品
#聚类查看分组信息和有无异常样品。如果这确实是由有意义的生物变化引起的，也
#可以使用下面的经验power值。
if (is.na(power)){
  power = ifelse(nSamples<20, ifelse(type == "unsigned", 9, 18),
                 ifelse(nSamples<30, ifelse(type == "unsigned", 8, 16),
                        ifelse(nSamples<40, ifelse(type == "unsigned", 7, 14),
                               ifelse(type == "unsigned", 6, 12))  ) )}

#2.2共表达网络构建与模块划分
#构建共表达网络并划分基因模块
##一步法网络构建：One-step network construction and module detection##
# power: 上一步计算的软阈值
# maxBlockSize: 计算机能处理的最大模块的基因数量 (默认5000)；
#  4G内存电脑可处理8000-10000个，16G内存电脑可以处理2万个，32G内存电脑可
#  以处理3万个
#  计算资源允许的情况下最好放在一个block里面。
# corType: pearson or bicor
# numericLabels: 返回数字而不是颜色作为模块的名字，后面可以再转换为颜色
# saveTOMs：最耗费时间的计算，存储起来，供后续使用
# mergeCutHeight: 合并模块的阈值，越大模块越少
nGenes = ncol(datExpr0)
net = blockwiseModules(datExpr0, power = power, maxBlockSize = 8658,
                       TOMType = "unsigned", minModuleSize = 30,
                       reassignThreshold = 0, mergeCutHeight = 0.25,
                       numericLabels = TRUE, pamRespectsDendro = FALSE,
                       saveTOMs=TRUE,
                       saveTOMFileBase ="femaleMouseTOM",
                       verbose = 3)
#统计模块的基因数
table(net$colors)
save(net,file = "net.RData")

2.3 绘制模块聚类树

load(file = "net.RData")
#将数字标签转换为颜色
mergedColors <- labels2colors(net$colors)
#查看聚类树
net$dendrograms[[1]]
#获取基因序号（最后6个）
tail(net$blockGenes[[1]])
#绘制聚类树和基因模块组合图
plotDendroAndColors(net$dendrograms[[1]],mergedColors[net$blockGenes[[1]]],
                    "Module colors",
                    dendroLabels = FALSE,hang=0.03,
                    addGuide = TRUE,guideHang = 0.05)
#提取模块标签和颜色
moduleLabels <- net$colors
moduleColors <- labels2colors(net$colors)
#存储模块特征基因到变量MEs中
MEs <- net$MEs
#存储聚类树到geneTree变量中
geneTree<-net$dendrograms[[1]]
#保存内存中的变量便于后续分析
save(MEs,moduleLabels,moduleColors,geneTree,file = "WGCNA_02_networkConstruction-auto.RData")

3 模块与性状关联分析

#模块与表型关联
#导入中间数据2
load(file = "WGCNA_01_data.RData")
load(file = "WGCNA_02_networkConstruction-auto.RData")
ls()
#明确基因数和样本数
nGenes<-ncol(datExpr0)
nSamples<-nrow(datExpr0)
#查看前六个基因对应的模块颜色
head(moduleColors)
#重新计算模块特征基因
MEs0=moduleEigengenes(datExpr0,moduleColors)$eigengenes
#对模块进行重新排序，使相关系数接近的模块彼此挨着
MEs=orderMEs(MEs0)
dim(MEs)
dim(datExpr0)
#计算两个矩阵(模块特征基因表达量与表型数据)的相关性
moduleTraitCor = cor(MEs,traitData,use = "p")
#计算pvalue
moduleTraitPvalue=corPvalueStudent(moduleTraitCor,nSamples)
#生成文本标签矩阵，合并相关系数和p值
cortext<-signif(moduleTraitCor,2)
pvaluetext<-signif(moduleTraitPvalue,1)
#tips:signif(pi,2)为3.1，用于取有效数字

#合并变量
textMatrix =paste(cortext,"\n(",pvaluetext,")",sep = "")
#将向量转成矩阵
dim(textMatrix)=dim(moduleTraitCor)
#查看标签矩阵的前4行前4列
textMatrix[1:4,1:4]
#设置绘图区边距大小（下，作，上，右）
par(mar=c(8,9,4,3))
#绘制模块与表型的相关性热图
labeledHeatmap(Matrix = moduleTraitCor,
               xLabels = names(datExpr0),
               yLabels = names(MEs),
               ySymbols = names(MEs),
               colorLabels = FALSE,
               colors = blueWhiteRed(100),
               textMatrix = textMatrix,
               setStdMargins = FALSE,
               cex.text = 0.5,
               zlim=c(-1,1),
               main=paste("Module-trait relationships"))
