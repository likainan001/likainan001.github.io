GO&KEGG富集分析（二）--非模式物种OrgDB的构建

Clusterprofile&OrgDB

ClusterProfiler是一个功能强大的R包，同时支持GO和KEGG的富集分析，而且可视化功能非常的优秀，那么怎么利用这个R包来进行富集分析。
进行分析时，首先就要求注释信息。Bioconductor上提供了以下19个物种的Org类型的包，包含了这些模式物种的GO及KEGG注释信息


对于以上19个物种，只需要安装对应的org包，clusterProfile就会自动从中获取GO注释信息，我们只需要差异基因的列表就可以了，使用起来非常方便。但是没有Orgdb的物种，就需要自己构建或者利用他人上传的数据，这里建议自己构建，不存在ID的错误问题。

构建非模式物种OrgDB

使用eggNOG-mapper结果进行构建：

 2#install.packages("")
 3library(tidyverse)

13options(stringsAsFactors = F)
14#顺手设置一下options
15emapper <- read.table("out.emapper.annotations.txt",header = TRUE, sep = "\t")
16#替换空格值
17emapper[emapper==""]<-NA
18
19gene_info <- emapper %>% dplyr::select(GID = query, GENENAME = Preferred_name) %>% na.omit()
20gos <- emapper %>% dplyr::select(query, GOs) %>% na.omit()
21
22gene2go = data.frame(GID = character(), GO = character(), EVIDENCE = character())
23
24for (row in 1:nrow(gos)) {
25  the_gid <- gos[row,"query"][[1]]
26  the_gos <- str_split(gos[row,"GOs"], ",", simplify = FALSE)[[1]]
27  df_temp <- data_frame(GID = rep(the_gid, length(the_gos)),
28                        GO = the_gos,
29                        EVIDENCE = rep("IDA", length(the_gos)))
30  gene2go <- rbind(gene2go, df_temp)}
31
32gene2go$GO[gene2go$GO=="-"]<-NA
33gene2go<-na.omit(gene2go)
34
35#Step3:提取KEGG信息
36
37gene2ko <- emapper %>% dplyr::select(GID = query, Ko = KEGG_ko) %>% na.omit()
38
39gene2ko$Ko <- gsub("ko:","",gene2ko$Ko)
40
41getwd()
42
43
44update_kegg <- function(json = "ko00001.json") {
45  pathway2name <- tibble(Pathway = character(), Name = character())
46  ko2pathway <- tibble(Ko = character(), Pathway = character())
47  kegg <- fromJSON(json)
48  for (a in seq_along(kegg[["children"]][["children"]])) {
49    A <- kegg[["children"]][["name"]][[a]]
50    for (b in seq_along(kegg[["children"]][["children"]][[a]][["children"]])) {
51      B <- kegg[["children"]][["children"]][[a]][["name"]][[b]] 
52      for (c in seq_along(kegg[["children"]][["children"]][[a]][["children"]][[b]][["children"]])) {
53        pathway_info <- kegg[["children"]][["children"]][[a]][["children"]][[b]][["name"]][[c]]
54        pathway_id <- str_match(pathway_info, "ko[0-9]{5}")[1]
55        pathway_name <- str_replace(pathway_info, " \\[PATH:ko[0-9]{5}\\]", "") %>% str_replace("[0-9]{5} ", "")
56        pathway2name <- rbind(pathway2name, tibble(Pathway = pathway_id, Name = pathway_name))
57        kos_info <- kegg[["children"]][["children"]][[a]][["children"]][[b]][["children"]][[c]][["name"]]
58        kos <- str_match(kos_info, "K[0-9]*")[,1]
59        ko2pathway <- rbind(ko2pathway, tibble(Ko = kos, Pathway = rep(pathway_id, length(kos))))}}}
60  save(pathway2name, ko2pathway, file = "kegg_info.RData")}
61# 调用函数后在本地创建kegg_info.RData文件
62update_kegg()
63# 载入kegg_info.RData文件
64load(file = "kegg_info.RData")
65
66gene2pathway <- gene2ko %>% left_join(ko2pathway, by = "Ko") %>% dplyr::select(GID, Pathway) %>% na.omit()
67
68tax_id = "NCBI查询"

78               author = "name <邮箱>",  #修改为你的名字和邮箱
79               outputDir = ".",  #输出文件位置
80               tax_id=tax_id,  #你在NCBI上查并定义的id

82               species=species,  #你在NCBI上查并定义的种名
83               goTable="go")
